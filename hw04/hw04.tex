\subsubsection*{(a)}
$$(x^3+x)\times(x+1)$$
\begin{equation}
\begin{split}
&=(x^4+x^2+x^3+x)\mod(x^4+x^3+1)\\
&\because \ x^4+x^3+1=0\\
&\therefore x^4=x^3+1\\
&=x^3+1+x^2+x^3+x\\
&=x^2+x+1
\end{split}
\end{equation}
